Install the following in /etc/systemd/system/jetty.service
Adjust the distribution and link references to suit


[Unit]
Description=Jetty
After=network-online.target

[Service]
Type=simple
User=jetty
Group=jetty
ExecStart= /bin/java -Xms64m -Xmx1g -Xss2M -Djava.awt.headless=true \
          -Djava.io.tmpdir=/var/www/temp \
          -Djetty.logs=/var/www/jetty/logs \
          -DGEOWEBCACHE_CACHE_DIR=/data/gwc \
          -DGEOSERVER_DATA_DIR=/data/gs_dir \
          -Dgeonetwork.dir=/data/gn_dir \
          -Dgeonetwork.lucene.dir=/data/gn_dir/index \
          -Dgeonetwork.log=/var/www/jetty/logs/ \
          -Djeeves.filecharsetdetectandconvert=enabled \
          -Dmime-mappings=/var/www/jetty/webapps/geonetwork/WEB-INF/mime-types.properties \
          -DSTOP.PORT=8079 -DSTOP.KEY=geonetwork \
          -jar /usr/share/java/jetty-distribution-9.4.35.v20201120/start.jar
ExecStop=/bin/kill ${MAINPID}
SuccessExitStatus=143

[Install]
WantedBy=multi-user.target
